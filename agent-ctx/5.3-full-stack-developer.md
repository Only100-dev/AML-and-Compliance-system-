## Pattern Applied (All Files)
```tsx
import { useSession } from 'next-auth/react';
import { getCurrency, formatCurrencyAmount } from '@/lib/gcc';

// Inside component:
const { data: session } = useSession();
const sessionUser = session?.user as { jurisdictionCode?: string | null } | undefined;
const jurisdictionCode = sessionUser?.jurisdictionCode || 'AE';
const activeCurrency = getCurrency(jurisdictionCode);

// Display:
formatCurrencyAmount(amount, activeCurrency)  // replaces formatAED(amount)
`Label (${activeCurrency})`                   // replaces "Label (AED)"
```
